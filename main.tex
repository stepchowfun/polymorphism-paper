    \begin{mdframed}

      \begin{prooftree}
          \AxiomC{$c$ is a constant of type $P$}
        \RightLabel{(\textsc{T-Const})}
        \UnaryInfC{$\Gamma \vdash c : P$}
      \end{prooftree}

      \begin{prooftree}
          \AxiomC{$x : T \in \Gamma$}
        \RightLabel{(\textsc{T-Var})}
        \UnaryInfC{$\Gamma \vdash x : T$}
      \end{prooftree}

      \begin{prooftree}
          \AxiomC{$\Gamma, x : T_1 \vdash t_2 : T_2$}
        \RightLabel{(\textsc{T-Abs})}
        \UnaryInfC{$\Gamma \vdash \labs{x : T_2}{T_2} : T_1 \rightarrow T_2$}
      \end{prooftree}

      \begin{prooftree}
          \AxiomC{$\Gamma \vdash t_1 : X \rightarrow Y$}
          \AxiomC{$\Gamma \vdash t_2 : X$}
        \RightLabel{(\textsc{T-App})}
        \BinaryInfC{$\Gamma \vdash \lapp{t_1}{t_2} : Y$}
      \end{prooftree}

      \begin{prooftree}
          \AxiomC{$\Gamma \vdash t : T$}
        \RightLabel{(\textsc{T-TAbs})}
        \UnaryInfC{$\Gamma \vdash \labs{X}{t} : \tforall{X}{T}$}
      \end{prooftree}

      \begin{prooftree}
          \AxiomC{$\Gamma \vdash t : \tforall{X}{T_1}$}
        \RightLabel{(\textsc{T-TApp})}
        \UnaryInfC{$\Gamma \vdash \lapp{t}{T_2} : \lsub{T_1}{X}{T_2}$}
      \end{prooftree}

    \end{mdframed}
    \caption{Typing rules for System F}
    \label{fig:system_f_typing_rules}
  \end{figure}
